### 1.2 核心设计原则

1. **Python原生性能优化**：使用`numpy`进行数值计算，`dataclasses`定义数据结构，`asyncio`处理异步
2. **内存高效管理**：使用预分配的`deque`和`list`缓冲区，避免频繁内存分配
3. **事件驱动状态机**：状态转换由Tick事件触发，**严禁**使用定时器驱动
4. **单向数据流**：策略层只发命令，不等待执行结果，执行状态由独立监控协程跟踪
5. **小资金优化**：专为300U-10万U设计的风险参数（1%单笔风险，2 tick止损，6 tick止盈）

### 3.1 Range Bar生成器（Algorithm 1 - Python实现）

**输入**: `NormalizedTick`流
**输出**: `RangeBar`流
**触发条件**: 价格物理位移达到`tick_range` (20 Ticks)

```python
class RangeBarGenerator:
    """Range Bar生成器（Python实现）"""

    def __init__(self, config: RangeBarConfig):
        self.config = config
        self.current_bar: Optional[RangeBar] = None
        self.open_px_base: float = 0.0
        self.bar_history: deque = deque(maxlen=config.max_bar_history)

    def on_tick(self, tick: NormalizedTick) -> Optional[RangeBar]:
        """处理单个Tick，可能返回闭合的Bar"""
        if self.current_bar is None:
            # 初始化新Bar
            self.current_bar = RangeBar(
                open_ts=tick.ts,
                open_px=tick.px,
                high_px=tick.px,
                low_px=tick.px,
                close_px=tick.px,
                total_buy_vol=tick.sz if tick.side == 1 else 0.0,
                total_sell_vol=tick.sz if tick.side == -1 else 0.0,
                delta=tick.sz if tick.side == 1 else -tick.sz,
                tick_count=1,
                is_closed=False
            )
            self.open_px_base = tick.px
            return None

        # 更新当前Bar
        self.current_bar.high_px = max(self.current_bar.high_px, tick.px)
        self.current_bar.low_px = min(self.current_bar.low_px, tick.px)
        self.current_bar.close_px = tick.px

        if tick.side == 1:
            self.current_bar.total_buy_vol += tick.sz
        else:
            self.current_bar.total_sell_vol += tick.sz

        self.current_bar.delta = self.current_bar.total_buy_vol - self.current_bar.total_sell_vol
        self.current_bar.tick_count += 1

        # 检查是否触发闭合条件
        displacement = abs(tick.px - self.open_px_base) / self.config.tick_range
        if displacement >= 1.0:  # 达到20个Tick位移
            return self._close_bar_and_emit(tick)

        return None

    def _close_bar_and_emit(self, overflow_tick: NormalizedTick) -> RangeBar:
        """闭合当前Bar，处理溢出Tick"""
        if self.current_bar is None:
            return None

        self.current_bar.is_closed = True
        completed_bar = self.current_bar
        self.bar_history.append(completed_bar)

        # 新开Bar，处理溢出
        self.current_bar = RangeBar(
            open_ts=overflow_tick.ts,
            open_px=overflow_tick.px,
            high_px=overflow_tick.px,
            low_px=overflow_tick.px,
            close_px=overflow_tick.px,
            total_buy_vol=overflow_tick.sz if overflow_tick.side == 1 else 0.0,
            total_sell_vol=overflow_tick.sz if overflow_tick.side == -1 else 0.0,
            delta=overflow_tick.sz if overflow_tick.side == 1 else -overflow_tick.sz,
            tick_count=1,
            is_closed=False
        )
        self.open_px_base = overflow_tick.px

        return completed_bar
```

### 3.2 KDE与LVN提取引擎（Algorithm 2 - Python实现）

**输入**: `RangeBar`序列（脉冲波段）
**输出**: `LVN_Zone`数组（最多3个区域）

```python
import numpy as np
from scipy import stats
from typing import List, Tuple


class KDEEngine:
    """KDE引擎（Python实现，使用numpy/scipy优化）"""

    def __init__(self, config: KDEEngineConfig):
        self.config = config

    def extract_lvn(self, slice_data: ImpulseSlice, tick_size: float = 0.01) -> List[Tuple[float, float]]:
        """
        提取LVN区域

        Args:
            slice_data: 脉冲波切片数据
            tick_size: 最小价格变动单位

        Returns:
            List[Tuple[float, float]]: LVN区域列表 [(lower, upper), ...]
        """
        if len(slice_data.ticks) < self.config.min_slice_ticks:
            return []

        # Step 1: 提取价格和成交量数据
        prices = np.array([tick.px for tick in slice_data.ticks])
        volumes = np.array([tick.sz for tick in slice_data.ticks])

        # Step 2: 稳健Silverman带宽计算
        h = self._calculate_robust_bandwidth(prices)

        # Step 3: 构建加权KDE（使用高斯核）
        price_range = np.linspace(
            prices.min() - 5 * tick_size,
            prices.max() + 5 * tick_size,
            int((prices.max() - prices.min()) / tick_size) + 10
        )

        density = np.zeros_like(price_range)
        for price, volume in zip(prices, volumes):
            # 高斯核函数
            kernel = np.exp(-0.5 * ((price_range - price) / h) ** 2) / (h * np.sqrt(2 * np.pi))
            density += volume * kernel

        # Step 4: 寻找局部极小值（LVN候选）
        minima_indices = self._find_local_minima(density)

        # Step 5: 密度阈值过滤（<30%分位）
        density_threshold = np.percentile(density, self.config.lvn_density_percentile)
        lvn_zones = []

        for idx in minima_indices:
            if density[idx] < density_threshold:
                center_price = price_range[idx]
                lvn_zones.append((
                    center_price - tick_size,
                    center_price + tick_size
                ))

        # 最多返回3个，按密度升序排序
        lvn_zones = sorted(lvn_zones, key=lambda x: x[0])[:self.config.max_lvn_zones]

        return lvn_zones

    def _calculate_robust_bandwidth(self, prices: np.ndarray) -> float:
        if len(prices) < self.config.min_slice_ticks:  # 通常是100
            return None  # 数据不足，跳过本次KDE计算

        std_dev = np.std(prices)
        iqr = np.percentile(prices, 75) - np.percentile(prices, 25)
        n = len(prices)

        # 防止除以零
        if iqr == 0:
            iqr = std_dev * 0.5  # 备用方案

        # Silverman法则
        h = 0.9 * min(std_dev, iqr / 1.34) * (n ** -0.2)

        # 限制带宽范围（防止极端值）
        tick_size = 0.01
        h = max(h, tick_size * 2)  # 最小2个Tick
        h = min(h, (prices.max() - prices.min()) * 0.1)  # 最大10%范围

        return h

    def _find_local_minima(self, density: np.ndarray) -> List[int]:
        """寻找局部极小值点"""
        minima = []
        for i in range(1, len(density) - 1):
            if density[i] < density[i - 1] and density[i] < density[i + 1]:
                minima.append(i)
        return minima
```

### 3.3 Triple-A状态机（Algorithm 3 - Python实现）

**状态定义**:

```python
class TripleAState:
    """Triple-A状态枚举"""
    IDLE = "IDLE"  # 休眠：价格不在任何LVN区域内
    MONITORING_ABSORPTION = "MONITORING_ABSORPTION"  # 监测中：价格进入LVN，监测努力vs结果
    ABSORPTION_CONFIRMED = "ABSORPTION_CONFIRMED"  # 吸收确认：检测到高CVD低位移
    ACCUMULATING = "ACCUMULATING"  # 积累中：波动压缩+Tick密度上升
    AGGRESSION_TRIGGERED = "AGGRESSION_TRIGGERED"  # 攻击触发：满足所有开仓条件
```

**状态转换图**:

```
                   [价格进入LVN区域]
    IDLE ──────────────────────────────────────► MONITORING_ABSORPTION
         ◄──────────────────────────────────────┘ [价格离开LVN区域]
                          (reset)

    MONITORING_ABSORPTION ──[ effort>2σ && |result|<2ticks ]──► ABSORPTION_CONFIRMED
                                    │
                                    │ [波动>3ticks 或 超时2min]
                                    ▼
                              ACCUMULATING ◄──────┘
                                    │
                                    │ [Tick数>200 && 波动<3ticks]
                                    ▼
                              AGGRESSION_TRIGGERED ──[大单气泡/失衡/CVD斜率]──► EXECUTE_LONG/SHORT
```

**详细判定逻辑**:

```python
class TripleAStateMachine:
    """Triple-A状态机（Python实现）"""

    def __init__(self, config: TripleAConfig):
        self.config = config
        self.ctx = TripleAContext()
        self.kde_engine = KDEEngine(KDEEngineConfig())
        self.range_bar_gen = RangeBarGenerator(RangeBarConfig())
        self.historical_cvd = deque(maxlen=1000)  # 历史CVD数据

    async def on_normalized_tick(self, tick: NormalizedTick):
        """处理标准化Tick（异步）"""
        self._update_context(tick)

        if self.ctx.state == TripleAState.IDLE:
            if self._is_price_in_lvn(tick.px):
                await self._transition_to(TripleAState.MONITORING_ABSORPTION, tick)

        elif self.ctx.state == TripleAState.MONITORING_ABSORPTION:
            # 检查双维度超时
            time_elapsed_ms = (tick.ts - self.ctx.entry_ts) / 1_000_000
            tick_timeout = self.ctx.tick_count_in_lvn >= self.config.absorption_window_ticks
            time_timeout = time_elapsed_ms > 30000  # 30秒强制超时

            if tick_timeout or time_timeout:
                if not (effort_ok and result_ok and efficiency_ok):
                    await self._transition_to(TripleAState.IDLE, tick)
                    return
            # 累积努力与结果数据
            self.ctx.effort_cvd += tick.side * tick.sz
            self.ctx.result_delta_px = tick.px - self.ctx.entry_px
            self.ctx.tick_count_in_lvn += 1

            # 检查是否满足吸收条件
            if self.ctx.tick_count_in_lvn >= self.config.absorption_window_ticks:
                cvd_mean = self._get_historical_cvd_mean(self.config.absorption_window_ticks)
                cvd_std = self._get_historical_cvd_std(self.config.absorption_window_ticks)
                zscore = (self.ctx.effort_cvd - cvd_mean) / cvd_std if cvd_std > 0 else 0

                effort_ok = abs(zscore) > self.config.effort_zscore_threshold
                result_ok = abs(self.ctx.result_delta_px) <= self.config.max_result_px_deviation
                efficiency_ok = abs(
                    self.ctx.result_delta_px / max(self.ctx.effort_cvd, 1e-8)) > self.config.cvd_efficiency_min

                if effort_ok and result_ok and efficiency_ok:
                    await self._transition_to(TripleAState.ABSORPTION_CONFIRMED, tick)
                elif not self._is_price_in_lvn(tick.px):
                    await self._transition_to(TripleAState.IDLE, tick)

        elif self.ctx.state == TripleAState.ABSORPTION_CONFIRMED:
            # 监测积累条件
            self.ctx.acc_tick_count += 1
            self._update_accumulation_volatility(tick.px)

            range_ok = self._get_current_range_ticks() <= self.config.accumulation_max_range_ticks
            density_ok = self.ctx.acc_tick_count >= self.config.accumulation_min_tick_count
            timeout = (tick.ts - self.ctx.entry_ts) > self.config.accumulation_timeout_ms * 1_000_000

            if range_ok and density_ok:
                await self._transition_to(TripleAState.ACCUMULATING, tick)
            elif self._get_current_range_ticks() > self.config.accumulation_max_range_ticks or timeout:
                await self._transition_to(TripleAState.IDLE, tick)

        elif self.ctx.state == TripleAState.ACCUMULATING:
            # 监测攻击信号
            if self._detect_aggression(tick):
                self.ctx.trigger_px = tick.px
                self.ctx.aggression_sz = self._calculate_bubble_size(tick)
                await self._transition_to(TripleAState.AGGRESSION_TRIGGERED, tick)
            elif not self._is_price_in_lvn(tick.px):
                await self._transition_to(TripleAState.IDLE, tick)

        elif self.ctx.state == TripleAState.AGGRESSION_TRIGGERED:
            # 已发送交易指令，等待执行层确认
            # 此状态下不再处理新Tick，由RiskManager接管
            pass

    def _detect_aggression(self, tick: NormalizedTick) -> bool:
        """攻击信号检测"""
        # 1. 大单气泡检测（动态阈值）
        bubble_threshold = self._get_percentile_volume(self.config.aggression_print_percentile)
        bubble_ok = tick.sz > bubble_threshold

        # 2. 足迹失衡检测（3档连续300%失衡）
        imbalance_ok = self._check_footprint_imbalance(
            tick,
            self.config.aggression_imbalance_ratio,
            self.config.aggression_stacked_levels
        )

        # 3. CVD斜率确认
        current_slope = self._calculate_cvd_slope(tick)
        slope_ok = abs(current_slope) > self.config.aggression_cvd_slope_threshold

        # 方向一致性检查（与宏观状态一致）
        direction_ok = (tick.side == self._get_macro_direction())

        return (bubble_ok or imbalance_ok) and slope_ok and direction_ok

    async def _transition_to(self, new_state: str, tick: NormalizedTick):
        """状态转换"""
        old_state = self.ctx.state
        self.ctx.state = new_state
        self.ctx.entry_ts = tick.ts
        self.ctx.entry_px = tick.px

        # 状态特定的初始化
        if new_state == TripleAState.MONITORING_ABSORPTION:
            self.ctx.effort_cvd = 0.0
            self.ctx.result_delta_px = 0.0
            self.ctx.tick_count_in_lvn = 0

        elif new_state == TripleAState.ABSORPTION_CONFIRMED:
            self.ctx.acc_tick_count = 0
            self.ctx.acc_volatility = 0.0

        elif new_state == TripleAState.AGGRESSION_TRIGGERED:
            # 触发交易执行
            await self._execute_trade_signal(tick)
```

```python
@dataclass
class PositionSizingResult:
    """仓位计算结果"""
    qty: float  # 合约数量
    stop_px: float  # 止损价格（物理锚点）
    take_profit_px: float  # 止盈价格
    breakeven_px: float  # 保本价格


class RiskManager:
    """风险管理器（小资金优化版）"""

    def __init__(self, config: RiskManagementConfig):
        self.config = config
        self.daily_pnl = 0.0
        self.daily_trades = 0

    def calculate_position(self, entry_px: float, trigger_px: float, equity: float,
                           direction: int) -> PositionSizingResult:
        """
        计算仓位大小（小资金优化）

        Args:
            entry_px: 入场价格
            trigger_px: 触发价格
            equity: 账户权益
            direction: 1=多单, -1=空单

        Returns:
            PositionSizingResult: 仓位计算结果
        """
        # 检查日损失限制
        if self.daily_pnl < -equity * 0.05:  # 日损失超过5%
            return PositionSizingResult(0.0, 0.0, 0.0, 0.0)

        # 风险金额 = equity * 1%（小资金高风险回报比）
        risk_amount = equity * self.config.risk_per_trade_pct

        # 止损距离 = 2个Tick（小资金快速止损）
        stop_distance_ticks = self.config.stop_buffer_ticks
        stop_distance = stop_distance_ticks * 0.01  # ETH tick_size=0.01

        # 止盈距离 = 6个Tick（盈亏比1:3）
        take_profit_distance_ticks = self.config.breakeven_trigger_ticks
        take_profit_distance = take_profit_distance_ticks * 0.01

        # 合约数量 = risk_amount / stop_distance
        qty = round(risk_amount / stop_distance, 2)  # 精确到2位小数

        # 计算价格
        if direction == 1:  # 多单
            stop_px = trigger_px - stop_distance
            take_profit_px = trigger_px + take_profit_distance
            breakeven_px = entry_px + 0.01  # 1个Tick保本
        else:  # 空单
            stop_px = trigger_px + stop_distance
            take_profit_px = trigger_px - take_profit_distance
            breakeven_px = entry_px - 0.01  # 1个Tick保本

        return PositionSizingResult(qty, stop_px, take_profit_px, breakeven_px)

    def update_daily_pnl(self, pnl: float):
        """更新日盈亏"""
        self.daily_pnl += pnl
        self.daily_trades += 1
```

```python
class CVDTrailingGuard:
    """CVD保本追踪器（Python异步实现）"""

    def __init__(self):
        self.entry_px = 0.0
        self.side = 0
        self.qty = 0.0
        self.highest_cvd_since_entry = 0.0
        self.highest_px_since_entry = 0.0
        self.current_cvd_getter = None  # 获取当前CVD的回调函数

    async def on_position_opened(self, entry_px: float, side: int, qty: float, cvd_getter):
        """仓位开立"""
        self.entry_px = entry_px
        self.side = side
        self.qty = qty
        self.current_cvd_getter = cvd_getter
        self.highest_cvd_since_entry = await cvd_getter()
        self.highest_px_since_entry = entry_px

    async def on_tick(self, tick: NormalizedTick):
        """处理Tick进行保本追踪"""
        if self.side == 1 and tick.px > self.entry_px:  # 多单浮盈中
            current_cvd = await self.current_cvd_getter()
            self.highest_cvd_since_entry = max(self.highest_cvd_since_entry, current_cvd)
            self.highest_px_since_entry = max(self.highest_px_since_entry, tick.px)

            # 检测CVD背离：价格新高但CVD未新高
            price_new_high = tick.px > self.highest_px_since_entry * 0.998  # 0.2%容差
            cvd_not_new_high = current_cvd < self.highest_cvd_since_entry * 0.995  # 0.5%容差

            if price_new_high and cvd_not_new_high:
                # 触发保本：将止损移至保本价
                return True  # 需要调整止损

        return False  # 不需要调整
```
